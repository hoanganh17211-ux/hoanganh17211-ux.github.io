// Hàm tạo hạt rơi toàn màn hình function createSnowflake() { const flake = document.createElement('i'); // Bạn có thể thay 'fa-snowflake' thành 'fa-heart' nếu muốn rơi trái tim flake.className = 'fa-solid fa-snowflake snowflake'; flake.style.left = Math.random() * 100 + 'vw'; flake.style.opacity = Math.random(); flake.style.fontSize = Math.random() * 10 + 10 + 'px'; const duration = Math.random() * 3 + 2; // Rơi trong 2-5 giây flake.style.animationDuration = duration + 's'; document.body.appendChild(flake); setTimeout(() => { flake.remove(); }, duration * 1000); } // Tạo hạt rơi liên tục mỗi 200ms setInterval(createSnowflake, 200);
